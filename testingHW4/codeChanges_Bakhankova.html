public Response returnMoney() {¶

if (mode == Mode.ADMINISTERING) return Response.ILLEGAL_OPERATION;¶

¶

if (balance == 0) {¶ return Response.OK;¶

} else if (balance > coins1 * coinval1 + coins2 * coinval2) {¶ return Response.TOO_BIG_CHANGE;¶

} else if (balance > coins2 * coinval2) {¶

// using coinval1 == 1¶

coins1 -= (balance - coins2 * coinval2);¶ coins2 = 0;¶

balance = 0;¶

¶

return Response.OK;¶

} else if (balance % coinval2 == 0) {¶ coins2 -= (balance / coinval2);¶ balance = 0;¶

¶

return Response.OK;¶

} else if (coins1 == 0) {¶

// using coinval1 == 1¶

return Response.UNSUITABLE_CHANGE;¶

} else {¶

// using coinval1 == 1¶

coins2 -= (balance / coinval2);¶ coins1--;¶

balance = 0;¶

¶

return Response.OK;¶

}¶
